Smart Tracker Project

Project Overview

1. Problem Statement

A Smart Tracker for Personal Belongings is an app or system that helps users track and monitor their personal items (like keys, wallets, or smartphones) in real-time. The system includes features to:

  • üìç Track items using tags or unique identifiers.
  • üîî Notify the user if an item is out of range or lost.
  • ‚ö° Ensure efficient searching and updating of item statuses.

2. Challenges

  • Real-time tracking of items using tags (RFID, GPS, etc.).
  • Efficient search and lookup to identify specific items quickly.
  • Update item status in real time, including lost or found status.
  • Data structure management for tracking multiple items and locations.

3. Solutions Using Algorithms and Data Structures

  • Tracking Items: Use a Hash Table for storing and quickly retrieving item details based on unique identifiers (e.g., item ID).
  • Updating Item Status: Use a Queue to store real-time updates on item statuses (e.g., item found, item lost).
  • Efficient Searching: Use a Binary Search Tree (BST) or AVL Tree to store and efficiently search for items in the system (based on unique identifiers or other properties).
  • Item Range Monitoring: Use a Trie to store location-based data (e.g., nearby items or places) for efficient range checking.

#include <iostream>
#include <unordered_map>
#include <queue>
#include <string>
#include <unordered_set>
#include <vector>

using namespace std;

// Item Structure
struct Item {
    string name;
    string status; // "lost", "found", "in range", "out of range"
    string location;
};

// Hash Table for storing item details
class ItemTracker {
private:
    unordered_map<string, Item> itemMap;

public:
    void addItem(string id, string name, string status, string location) {
        itemMap[id] = {name, status, location};
    }

    void updateItemStatus(string id, string newStatus) {
        if (itemMap.find(id) != itemMap.end()) {
            itemMap[id].status = newStatus;
        }
    }

    Item* getItem(string id) {
        if (itemMap.find(id) != itemMap.end()) {
            return &itemMap[id];
        }
        return nullptr;
    }

    void printItemDetails(string id) {
        Item* item = getItem(id);
        if (item) {
            cout << "Item: " << item->name << "\nStatus: " << item->status << "\nLocation: " << item->location << endl;
        } else {
            cout << "Item not found." << endl;
        }
    }
};

// Queue for real-time updates on item statuses
class UpdateQueue {
private:
    queue<string> updateQueue;

public:
    void addUpdate(string update) {
        updateQueue.push(update);
    }

    void processUpdates() {
        while (!updateQueue.empty()) {
            cout << "UPDATE: " << updateQueue.front() << endl;
            updateQueue.pop();
        }
    }
};

// Binary Search Tree (BST) for efficient searching of items
class BST {
private:
    struct TreeNode {
        string id;
        Item item;
        TreeNode* left;
        TreeNode* right;

        TreeNode(string id, Item item) : id(id), item(item), left(nullptr), right(nullptr) {}
    };

    TreeNode* root;

    TreeNode* insert(TreeNode* root, string id, Item item) {
        if (root == nullptr) {
            return new TreeNode(id, item);
        }
        if (id < root->id) {
            root->left = insert(root->left, id, item);
        } else if (id > root->id) {
            root->right = insert(root->right, id, item);
        }
        return root;
    }

    TreeNode* search(TreeNode* root, string id) {
        if (root == nullptr || root->id == id) {
            return root;
        }
        if (id < root->id) {
            return search(root->left, id);
        }
        return search(root->right, id);
    }

public:
    BST() : root(nullptr) {}

    void insert(string id, Item item) {
        root = insert(root, id, item);
    }

    Item* searchItem(string id) {
        TreeNode* result = search(root, id);
        if (result) {
            return &result->item;
        }
        return nullptr;
    }
};

// Trie for location-based management (tracking nearby items)
class LocationTrieNode {
public:
    unordered_map<char, LocationTrieNode*> children;
    bool isEndOfLocation;
    LocationTrieNode() {
        isEndOfLocation = false;
    }
};

class LocationTrie {
private:
    LocationTrieNode* root;

public:
    LocationTrie() {
        root = new LocationTrieNode();
    }

    void insert(string location) {
        LocationTrieNode* currentNode = root;
        for (char ch : location) {
            if (currentNode->children.find(ch) == currentNode->children.end()) {
                currentNode->children[ch] = new LocationTrieNode();
            }
            currentNode = currentNode->children[ch];
        }
        currentNode->isEndOfLocation = true;
    }

    bool search(string location) {
        LocationTrieNode* currentNode = root;
        for (char ch : location) {
            if (currentNode->children.find(ch) == currentNode->children.end()) {
                return false;
            }
            currentNode = currentNode->children[ch];
        }
        return currentNode->isEndOfLocation;
    }
};

// Main function simulating Smart Tracker for Personal Belongings
int main() {
    // Initialize components
    ItemTracker tracker;
    UpdateQueue updateQueue;
    BST itemBST;
    LocationTrie locationTrie;

    // Add items to tracker
    tracker.addItem("ID001", "Wallet", "found", "Living Room");
    tracker.addItem("ID002", "Keys", "lost", "Bedroom");
    tracker.addItem("ID003", "Phone", "in range", "Kitchen");

    // Insert items into BST for searching
    itemBST.insert("ID001", {"Wallet", "found", "Living Room"});
    itemBST.insert("ID002", {"Keys", "lost", "Bedroom"});
    itemBST.insert("ID003", {"Phone", "in range", "Kitchen"});

    // Add some location data to the Trie
    locationTrie.insert("Living Room");
    locationTrie.insert("Bedroom");

    // Simulate real-time updates
    updateQueue.addUpdate("Item ID001 (Wallet) moved to Living Room");
    updateQueue.addUpdate("Item ID002 (Keys) moved to Bedroom");
    updateQueue.addUpdate("Item ID003 (Phone) moved to Kitchen");

    // Process real-time updates
    updateQueue.processUpdates();

    // Print item details
    tracker.printItemDetails("ID001");
    tracker.printItemDetails("ID002");

    // Search for an item in the BST
    Item* foundItem = itemBST.searchItem("ID003");
    if (foundItem) {
        cout << "Item found in BST: " << foundItem->name << " at " << foundItem->location << endl;
    } else {
        cout << "Item not found in BST." << endl;
    }

    // Search for a location in the Trie
    if (locationTrie.search("Living Room")) {
        cout << "Location 'Living Room' exists in the Trie." << endl;
    } else {
        cout << "Location 'Living Room' does not exist in the Trie." << endl;
    }

    return 0;
}
